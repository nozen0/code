package tuan07_QlTpham;

import java.util.Date;

// Lớp hàng thực phẩm kế thừa từ HangHoa
public class HangThucPham extends HangHoa {
    private String nhaCungCap; // Nhà cung cấp
    private Date ngaySanXuat; // Ngày sản xuất
    private Date ngayHetHan; // Ngày hết hạn

    public HangThucPham(String maHang, String tenHang, double donGia, int soLuongTon, String nhaCungCap, Date ngaySanXuat, Date ngayHetHan) {
        super(maHang, tenHang, donGia, soLuongTon); // Gọi constructor lớp cha
        this.nhaCungCap = nhaCungCap;
        this.ngaySanXuat = ngaySanXuat;
        this.ngayHetHan = ngayHetHan;
    }

    // Ghi đè phương thức toString() để hiển thị thông tin chi tiết về hàng thực phẩm
    @Override
    public String toString() {
        return "Mã hàng: " + maHang + ", Tên hàng: " + tenHang + ", Đơn giá: " + donGia + ", Số lượng tồn: " + soLuongTon +
                ", Nhà cung cấp: " + nhaCungCap + ", Ngày sản xuất: " + ngaySanXuat + ", Ngày hết hạn: " + ngayHetHan;
    }
}
