package tuan07_QlTpham;

import java.util.Date;

// Lớp hàng điện máy kế thừa từ HangHoa
public class HangDienMay extends HangHoa {
    private int baoHanh; // Thời gian bảo hành (tháng)
    private int congSuat; // Công suất (watt)

    public HangDienMay(String maHang, String tenHang, double donGia, int soLuongTon, int baoHanh, int congSuat) {
        super(maHang, tenHang, donGia, soLuongTon); // Gọi constructor lớp cha
        this.baoHanh = baoHanh;
        this.congSuat = congSuat;
    }

    // Ghi đè phương thức toString() để hiển thị thông tin chi tiết về hàng điện máy
    @Override
    public String toString() {
        return "Mã hàng: " + maHang + ", Tên hàng: " + tenHang + ", Đơn giá: " + donGia + ", Số lượng tồn: " + soLuongTon +
                ", Bảo hành: " + baoHanh + " tháng, Công suất: " + congSuat + " watt";
    }
}
