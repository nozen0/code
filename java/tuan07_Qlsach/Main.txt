package tuan07_Qlsach;

import java.util.List;

public class Main {
    public static void main(String[] args) {
        QuanLySach quanLy = new QuanLySach();
        
        // Thêm sách mẫu
        quanLy.themSach(new SachGiaoKhoa("SGK001", "01-01-2023", 50000, 10, "NXB K", "moi"));
        quanLy.themSach(new SachGiaoKhoa("SGK002", "02-02-2023", 40000, 5, "NXB A", "cu"));
        quanLy.themSach(new SachGiaoKhoa("SGK003", "03-03-2023", 60000, 8, "NXB K", "moi"));
        
        quanLy.themSach(new SachThamKhao("STK001", "01-04-2023", 80000, 7, "NXB B", 5000));
        quanLy.themSach(new SachThamKhao("STK002", "02-05-2023", 70000, 6, "NXB C", 6000));
        quanLy.themSach(new SachThamKhao("STK003", "03-06-2023", 90000, 9, "NXB D", 7000));
        
        // Tính tổng thành tiền
        double tongGiaoKhoa = quanLy.tinhTongThanhTienGiaoKhoa();
        double tongThamKhao = quanLy.tinhTongThanhTienThamKhao();
        
        // Lấy danh sách sách giáo khoa của một NXB
        String nxb = "NXB K";
        List<String> sachGiaoKhoaTheoNXB = quanLy.xuatSachGiaoKhoaTheoNXB(nxb);
        
        // Tìm sách có thành tiền cao nhất
        Sach maxSach = quanLy.timSachThanhTienCaoNhat();
        
        // Xuất dữ liệu (có thể hiển thị theo yêu cầu)
        hienThiKetQua(tongGiaoKhoa, tongThamKhao, nxb, sachGiaoKhoaTheoNXB, maxSach);
    }

    // Phương thức riêng để hiển thị dữ liệu
    private static void hienThiKetQua(double tongGiaoKhoa, double tongThamKhao, String nxb, List<String> sachNXB, Sach maxSach) {
        System.out.println("Tổng thành tiền sách giáo khoa: " + tongGiaoKhoa);
        System.out.println("Tổng thành tiền sách tham khảo: " + tongThamKhao);
        
        System.out.println("Sách giáo khoa của nhà xuất bản " + nxb + ":");
        for (String sach : sachNXB) {
            System.out.println(sach); // In ra thông tin sách
        }
        
        if (maxSach != null) {
            System.out.println("Sách có thành tiền cao nhất: " + maxSach); // In ra sách có thành tiền cao nhất
        }
    }
}
