package tuan07_Qlsach;

public class SachGiaoKhoa extends Sach {
    private String tinhTrang;

    public SachGiaoKhoa(String maSach, String ngayNhap, double donGia, int soLuong, String nhaXuatBan, String tinhTrang) {
        super(maSach, ngayNhap, donGia, soLuong, nhaXuatBan);
        this.tinhTrang = tinhTrang;
    }

    @Override
    public double tinhThanhTien() {
        return tinhTrang.equalsIgnoreCase("moi") ? soLuong * donGia : soLuong * donGia * 0.5;
    }

    @Override
    public String toString() {
        return super.toString() + ", Tình Trạng: " + tinhTrang + ", Thành Tiền: " + tinhThanhTien();
    }
}
