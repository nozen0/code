package tuan07_Qlsach;

public class SachThamKhao extends Sach {
    private double giaBia;

    public SachThamKhao(String maSach, String ngayNhap, double donGia, int soLuong, String nhaXuatBan, double giaBia) {
        super(maSach, ngayNhap, donGia, soLuong, nhaXuatBan);
        this.giaBia = giaBia;
    }

    @Override
    public double tinhThanhTien() {
        return donGia * soLuong + giaBia;
    }

    @Override
    public String toString() {
        return super.toString() + ", Giá Bìa: " + giaBia + ", Thành Tiền: " + tinhThanhTien();
    }
}
