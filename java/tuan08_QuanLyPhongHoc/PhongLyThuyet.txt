package tuan08_QuanLyPhongHoc;

// Lớp kế thừa từ Phonghoc, đại diện cho phòng học lý thuyết
public class PhongLyThuyet extends Phonghoc {
    private boolean hasMayChieu; // Biến kiểm tra xem phòng có máy chiếu hay không

    // Constructor
    public PhongLyThuyet(String maPhong, String dayNha, double dienTich, int soBongDen, boolean hasMayChieu) {
        super(maPhong, dayNha, dienTich, soBongDen);
        this.hasMayChieu = hasMayChieu;
    }

    // Getter và Setter
    public boolean isHasMayChieu() {
        return hasMayChieu;
    }

    public void setHasMayChieu(boolean hasMayChieu) {
        this.hasMayChieu = hasMayChieu;
    }

    // Kiểm tra phòng có đạt chuẩn không (số bóng đèn/dien tích >= 10 và có máy chiếu)
    @Override
    protected boolean checkDatChuan() {
        return soBongDen / dienTich >= 10 && hasMayChieu;
    }

    @Override
    public String toString() {
        return "Phong ly thuyet";
    }
}
