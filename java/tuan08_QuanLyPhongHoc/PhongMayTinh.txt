package tuan08_QuanLyPhongHoc;

// Lớp đại diện cho phòng máy tính, kế thừa từ Phonghoc
public class PhongMayTinh extends Phonghoc {
    private int soMayTinh; // Số lượng máy tính trong phòng

    // Constructor
    public PhongMayTinh(String maPhong, String dayNha, double dienTich, int soBongDen, int soMayTinh) {
        super(maPhong, dayNha, dienTich, soBongDen);
        this.soMayTinh = soMayTinh;
    }

    // Getter và Setter
    public int getSoMayTinh() {
        return soMayTinh;
    }

    public void setSoMayTinh(int soMayTinh) {
        this.soMayTinh = soMayTinh;
    }

    // Kiểm tra phòng đạt chuẩn nếu số bóng đèn/dien tích >= 10 và số máy tính/dien tích >= 1.5
    @Override
    protected boolean checkDatChuan() {
        return (soBongDen / dienTich >= 10) && (soMayTinh / dienTich >= 1.5);
    }

    @Override
    public String toString() {
        return "Phong may tinh";
    }
}
